**apps/api/src/__tests__/health.test.ts** — smoke test that does NOT require a real DB or Redis connection (tests the route structure only):
```typescript
import { describe, it, expect, beforeAll, afterAll } from 'vitest';
import { buildApp } from '../app.js';
import type { FastifyInstance } from 'fastify';

// Mock environment variables for test
process.env.DATABASE_URL = 'postgresql://test:test@localhost:5432/test';
process.env.REDIS_URL = 'redis://localhost:6379';
process.env.API_JWT_SECRET = 'test-secret-minimum-32-chars-long!!';
process.env.NODE_ENV = 'test';

describe('Health route', () => {
  let app: FastifyInstance;

  beforeAll(async () => {
    // We can't test real DB/Redis in unit tests — integration tests are Phase 2+
    // This smoke test just verifies the app builds and routes are registered
    app = await buildApp();
  });

  afterAll(async () => {
    await app.close();
  });

  it('registers the /v1/health route', async () => {
    // Check the route exists (will return 503 without real DB — that's fine)
    const response = await app.inject({
      method: 'GET',
      url: '/v1/health',
    });
    // Accept 200 (infra available) or 503 (infra unavailable in test env) — both mean route is registered
    expect([200, 503]).toContain(response.statusCode);
    const body = JSON.parse(response.body);
    expect(body).toHaveProperty('status');
    expect(body).toHaveProperty('db');
    expect(body).toHaveProperty('redis');
  });
});
```

Create `.github/workflows/ci.yml`:
```yaml
name: CI

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

concurrency:
  group: ${{ github.workflow }}-${{ github.ref }}
  cancel-in-progress: true

jobs:
  ci:
    name: Lint + Type-check + Test
    runs-on: ubuntu-latest
    timeout-minutes: 15

    steps:
      - name: Checkout
        uses: actions/checkout@v4
        with:
          fetch-depth: 2  # Turborepo needs previous commit for affected detection

      - name: Setup pnpm
        uses: pnpm/action-setup@v4
        with:
          version: 9.15.0

      - name: Setup Node.js 22
        uses: actions/setup-node@v4
        with:
          node-version: 22
          cache: 'pnpm'

      - name: Install dependencies
        run: pnpm install --frozen-lockfile

      - name: Lint
        run: pnpm turbo run lint

      - name: Type-check
        run: pnpm turbo run type-check

      - name: Test
        run: pnpm turbo run test
        env:
          # Smoke tests don't need real services — env vars prevent "missing" errors
          DATABASE_URL: postgresql://postgres:postgres@localhost:5432/parabuains_test
          REDIS_URL: redis://localhost:6379
          API_JWT_SECRET: ci-test-secret-minimum-32-chars-long!!
          NODE_ENV: test
```

Also create `.github/` directory structure if it doesn't exist and ensure the file is written to the correct path `.github/workflows/ci.yml`.

| Boundary | Description |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Information Disclosure | CI env vars | mitigate | Secrets (real API keys, prod DB) are NOT in ci.yml; only smoke test placeholder values used. Real secrets go in GitHub repository Secrets, referenced as `${{ secrets.KEY }}` when needed in Phase 2+ |
| T-05-02 | Denial of Service | CI pipeline | mitigate | `cancel-in-progress: true` prevents queue buildup on rapid pushes; `timeout-minutes: 15` prevents runaway CI jobs |

</threat_model>
