import type { NextConfig } from 'next';

const nextConfig: NextConfig = {
  // App Router is the default in Next.js 16 — no appDir flag needed
  // Pages Router is disabled (not creating pages/)
  experimental: {
    // typedRoutes: true, // Enable in Phase 3 when routes are established
  },
  // Proxy internal API calls to Fastify (BFF pattern)
  async rewrites() {
    return [
      {
        source: '/api/v1/:path*',
        destination: `${process.env.NEXT_PUBLIC_API_URL ?? 'http://localhost:3001'}/v1/:path*`,
      },
    ];
  },
};

export default nextConfig;
