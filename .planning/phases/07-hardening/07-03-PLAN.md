```typescript
import { NextResponse } from 'next/server';
import type { NextRequest } from 'next/server';
import { v4 as uuidv4 } from 'uuid'; // ou usar crypto.randomUUID()

export function middleware(request: NextRequest) {
  const nonce = Buffer.from(crypto.randomUUID()).toString('base64');

  const cspHeader = [
    `default-src 'self'`,
    `script-src 'self' 'nonce-${nonce}'`,
    `style-src 'self' 'unsafe-inline'`,   // Tailwind CSS requer unsafe-inline
    `img-src 'self' data: https://*.r2.dev https://*.cloudflare.com`,
    `connect-src 'self' ${process.env.NEXT_PUBLIC_API_URL ?? ''}`,
    `font-src 'self'`,
    `frame-ancestors 'none'`,
    `object-src 'none'`,
    `base-uri 'self'`,
    `form-action 'self'`,
  ].join('; ');

  // Preservar logica existente (auth, rate limit, etc.)
  const response = NextResponse.next({
    request: {
      headers: new Headers(request.headers),
    },
  });

  // Passar nonce para o app via header customizado (lido pelo RootLayout)
  response.headers.set('x-nonce', nonce);

  // Security headers
  response.headers.set('Content-Security-Policy', cspHeader);
  response.headers.set('X-Frame-Options', 'DENY');
  response.headers.set('X-Content-Type-Options', 'nosniff');
  response.headers.set('Referrer-Policy', 'strict-origin-when-cross-origin');
  response.headers.set('Permissions-Policy', 'camera=(), microphone=(), geolocation=()');

  if (process.env.NODE_ENV === 'production') {
    response.headers.set(
      'Strict-Transport-Security',
      'max-age=31536000; includeSubDomains; preload'
    );
  }

  return response;
}

export const config = {
  matcher: ['/((?!_next/static|_next/image|favicon.ico).*)'],
};
```

Exemplo de correcao:
```tsx
// ERRADO — sem label
<input type="text" placeholder="Buscar amigos..." />

// CORRETO — com aria-label
<input
  type="text"
  placeholder="Buscar amigos..."
  aria-label="Buscar amigos"
  id="friend-search"
/>
// OU com label associado
<label htmlFor="friend-search" className="sr-only">Buscar amigos</label>
<input id="friend-search" type="text" placeholder="Buscar amigos..." />
```

## Trust Boundaries

| Boundary | Descricao |
|----------|-----------|
