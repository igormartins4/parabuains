**resend-email-transport.ts**:
```typescript
import { Resend } from 'resend';
import { renderAsync } from '@react-email/components';
import type { INotificationTransport, NotificationPayload } from './types.js';
import { BirthdayReminderEmail, WallMessagePostedEmail, FriendshipAcceptedEmail } from '@parabuains/email-templates';

export class ResendEmailTransport implements INotificationTransport {
  private resend: Resend;
  private fromEmail = 'noreply@parabuains.com';

  constructor(apiKey: string = process.env.RESEND_API_KEY ?? '') {
    this.resend = new Resend(apiKey);
  }

  async send(payload: NotificationPayload): Promise<void> {
    const { subject, html } = await this.renderTemplate(payload);
    const result = await this.resend.emails.send({
      from: this.fromEmail,
      to: payload.recipientEmail,
      subject,
      html,
    });
    if (result.error) throw new Error(`Resend error: ${result.error.message}`);
  }

  private async renderTemplate(payload: NotificationPayload): Promise<{ subject: string; html: string }> {
    switch (payload.eventType) {
      case 'birthday-reminder': {
        const daysUntil = payload.reminderType === 'Dday' ? 0
          : payload.reminderType === 'D1' ? 1
          : payload.reminderType === 'D3' ? 3 : 7;
        const subject = daysUntil === 0
          ? `Hoje e aniversario de ${payload.subjectName}!`
          : `Aniversario de ${payload.subjectName} em ${daysUntil} dia(s)`;
        const html = await renderAsync(BirthdayReminderEmail({
          recipientName: payload.recipientName,
          birthdayPersonName: payload.subjectName,
          daysUntil,
          profileUrl: payload.profileUrl,
        }));
        return { subject, html };
      }
      case 'wall-message-posted': {
        const html = await renderAsync(WallMessagePostedEmail({
          recipientName: payload.recipientName,
          authorName: payload.subjectName,
          profileUrl: payload.profileUrl,
        }));
        return { subject: `${payload.subjectName} deixou uma mensagem no seu mural`, html };
      }
      case 'friendship-accepted': {
        const html = await renderAsync(FriendshipAcceptedEmail({
          recipientName: payload.recipientName,
          friendName: payload.subjectName,
          profileUrl: payload.profileUrl,
        }));
        return { subject: `${payload.subjectName} aceitou seu pedido de amizade`, html };
      }
    }
  }
}
```

**birthday-reminder-email.consumer.ts** — queue: 'birthday-reminders':
```typescript
import { Worker, type Job } from 'bullmq';
import type { DrizzleDB } from '../../db/index.js';
import { users, notificationLog, notificationPreferences } from '../../db/schema.js';
import { eq, and } from 'drizzle-orm';
import { NotificationTransportFactory } from '../transports/factory.js';
import type { ReminderType } from '../transports/types.js';

export interface BirthdayReminderJobData {
  recipientId: string;
  subjectId: string;
  reminderType: ReminderType;
  scheduledFor: string; // ISO timestamp
}

export function createBirthdayReminderEmailWorker(db: DrizzleDB, redisConnection: { host: string; port: number }) {
  return new Worker<BirthdayReminderJobData>('birthday-reminders', async (job) => {
    const { recipientId, subjectId, reminderType } = job.data;

    // Idempotency check FIRST
    const existing = await db.select().from(notificationLog)
      .where(and(
        eq(notificationLog.recipientId, recipientId),
        eq(notificationLog.subjectId, subjectId),
        eq(notificationLog.channel, 'email'),
        eq(notificationLog.reminderType, reminderType),
      )).limit(1);
    if (existing.length > 0) return { status: 'skipped' };

    // Fetch data from DB — NOT from job payload (no PII in Redis)
    const [recipient, subject] = await Promise.all([
      db.select().from(users).where(eq(users.id, recipientId)).limit(1),
      db.select().from(users).where(eq(users.id, subjectId)).limit(1),
    ]);
    if (!recipient[0] || !subject[0]) throw new Error(`User not found`);

    // Check notification preferences
    const prefs = await db.select().from(notificationPreferences)
      .where(and(eq(notificationPreferences.userId, recipientId), eq(notificationPreferences.channel, 'email')))
      .limit(1);
    if (prefs[0] && !prefs[0].enabled) return { status: 'skipped' };

    // Send
    const transport = NotificationTransportFactory.create('email');
    await transport.send({
      eventType: 'birthday-reminder',
      recipientId,
      subjectId,
      reminderType,
      recipientEmail: recipient[0].email,
      recipientName: recipient[0].displayName,
      subjectName: subject[0].displayName,
      profileUrl: `${process.env.NEXT_PUBLIC_API_URL ?? 'https://parabuains.com'}/${subject[0].username}`,
    });

    // Log AFTER successful send
    await db.insert(notificationLog).values({ recipientId, subjectId, channel: 'email', reminderType, status: 'sent' });
    return { status: 'sent' };
  }, {
    connection: redisConnection,
    concurrency: 5,
    defaultJobOptions: {
      attempts: 3,
      backoff: { type: 'exponential', delay: 5000 },
      removeOnComplete: { count: 100 },
      removeOnFail: { count: 50 },
    },
  });
}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| BullMQ job data -> Worker | Jobs contêm apenas IDs — nunca PII (email, nome). Dados buscados do DB em runtime. |
| Worker -> Resend API | RESEND_API_KEY apenas via env var, nunca em logs ou job payloads |
| API -> notification_preferences | Endpoint autenticado — usuario so altera suas proprias preferencias |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-01-01 | Information Disclosure | BullMQ job payload | mitigate | Payload contem apenas IDs; worker busca email/nome do DB em runtime — PII nunca exposto no Redis/Bull Board |
