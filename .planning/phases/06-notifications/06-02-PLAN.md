```typescript
export interface PushSubscription {
  id: string;
  userId: string;
  endpoint: string;

  authKey: string;
  userAgent: string | null;
  createdAt: Date;

export interface IpushSubscriptionRepository {

  deleteByEndpoint(endpoint: string): Promise<void>;

}
```

`save` uses `onConflictDoNothing` on the endpoint unique constraint (idempotent).

### Task 3 (auto): VapidTransport

Create `apps/api/src/modules/notifications/vapid.transport.ts`:

```typescript
import webpush from 'web-push';

export class VapidTransport {
  constructor(publicKey: string, privateKey: string, subject: string) {
    webpush.setVapidDetails(subject, publicKey, privateKey);
  }

  async sendPushNotification(
    subscription: { endpoint: string; p256dhKey: string; authKey: string },
    payload: { title: string; body: string; url?: string },
  ): Promise<'sent' | 'invalid_endpoint'>;
}
```

If `webpush.sendNotification()` throws with status 404 or 410, return `'invalid_endpoint'` (subscription expired — caller should delete it). Otherwise throw.

**Done criteria:** File created; compiles without error.

### Task 6 (auto): Service Worker

Create `apps/web/public/sw.js`:

```javascript
self.addEventListener('push', (event) => {
  const data = event.data?.json() ?? { title: 'Parabuains', body: '' };
  event.waitUntil(
    self.registration.showNotification(data.title, {
      body: data.body,
      icon: '/icon-192.png',
      data: { url: data.url ?? '/' },
    })
  );
});

self.addEventListener('notificationclick', (event) => {
  event.notification.close();
  event.waitUntil(
    clients.openWindow(event.notification.data.url)
  );
});
```

**Done criteria:** File created at `apps/web/public/sw.js`.

### Task 8 (auto): usePushNotifications hook

Create `apps/web/hooks/use-push-notifications.ts`:

```typescript
export function usePushNotifications(): {
  isSupported: boolean;          // 'serviceWorker' in navigator && 'PushManager' in window
  permission: NotificationPermission; // 'default' | 'granted' | 'denied'
  isSubscribed: boolean;
  subscribe(): Promise<void>;   // registers SW, subscribes, POSTs to BFF
  unsubscribe(): Promise<void>; // unsubscribes, DELETEs from BFF
}
```

`subscribe()`:
1. Register service worker (`/sw.js`)
2. Call `pushManager.subscribe({ userVisibleOnly: true, applicationServerKey: VAPID_PUBLIC_KEY })`
3. POST to `/api/push/subscribe` with endpoint + keys
4. Update `isSubscribed` state

`unsubscribe()`:
1. Get existing subscription
2. DELETE `/api/push/subscribe` with endpoint
3. Call `subscription.unsubscribe()`
4. Update `isSubscribed` state

Env: `NEXT_PUBLIC_VAPID_PUBLIC_KEY`

**Done criteria:** File created; compiles without error.
