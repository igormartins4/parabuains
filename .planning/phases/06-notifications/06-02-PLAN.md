```typescript
import webPush from 'web-push';
import type { INotificationTransport, NotificationPayload } from './types.js';

// VAPID setup — chamado uma vez na inicializacao do worker
if (process.env.VAPID_PUBLIC_KEY && process.env.VAPID_PRIVATE_KEY) {
  webPush.setVapidDetails(
    process.env.VAPID_EMAIL ?? 'mailto:admin@parabuains.com',
    process.env.VAPID_PUBLIC_KEY,
    process.env.VAPID_PRIVATE_KEY,
  );
}

export interface PushSubscriptionRecord {
  endpoint: string;
  p256dhKey: string;
  authKey: string;
}

export class VapidPushTransport implements INotificationTransport {
  // subscription buscada do DB pelo consumer e injetada no payload
  // O transport espera recipientPushSubscription no payload
  async send(payload: NotificationPayload & { pushSubscription?: PushSubscriptionRecord }): Promise<void> {
    if (!payload.pushSubscription) {
      throw new Error('VapidPushTransport: pushSubscription is required');
    }

    const { endpoint, p256dhKey, authKey } = payload.pushSubscription;

    const pushPayload = JSON.stringify({
      title: this.buildTitle(payload),
      body: this.buildBody(payload),
      url: payload.profileUrl,
      icon: '/icons/notification-icon.png',
      badge: '/icons/notification-badge.png',
    });

    await webPush.sendNotification(
      { endpoint, keys: { p256dh: p256dhKey, auth: authKey } },
      pushPayload,
    );
  }

  private buildTitle(payload: NotificationPayload): string {
    switch (payload.eventType) {
      case 'birthday-reminder':
        return payload.reminderType === 'Dday'
          ? `Hoje e aniversario de ${payload.subjectName}!`
          : `Aniversario de ${payload.subjectName} em breve`;
      case 'wall-message-posted':
        return `Nova mensagem no seu mural`;
      case 'friendship-accepted':
        return `${payload.subjectName} aceitou sua amizade`;
    }
  }

  private buildBody(payload: NotificationPayload): string {
    switch (payload.eventType) {
      case 'birthday-reminder':
        return `Acesse o perfil e deixe uma mensagem para ${payload.subjectName}!`;
      case 'wall-message-posted':
        return `${payload.subjectName} deixou uma mensagem para voce.`;
      case 'friendship-accepted':
        return `Agora voce pode ver os aniversarios de ${payload.subjectName}.`;
    }
  }
}
```

```javascript
// apps/web/public/sw.js
// Parabuains Service Worker — handles Web Push notifications

self.addEventListener('install', (event) => {
  self.skipWaiting();
});

self.addEventListener('activate', (event) => {
  event.waitUntil(self.clients.claim());
});

self.addEventListener('push', (event) => {
  if (!event.data) return;

  let data;
  try {
    data = event.data.json();
  } catch {
    data = { title: 'Parabuains', body: event.data.text(), url: '/' };
  }

  const options = {
    body: data.body ?? '',
    icon: data.icon ?? '/icons/notification-icon.png',
    badge: data.badge ?? '/icons/notification-badge.png',
    data: { url: data.url ?? '/' },
    requireInteraction: false,
    silent: false,
  };

  event.waitUntil(
    self.registration.showNotification(data.title ?? 'Parabuains', options)
  );
});

self.addEventListener('notificationclick', (event) => {
  event.notification.close();

  const targetUrl = event.notification.data?.url ?? '/';

  event.waitUntil(
    self.clients.matchAll({ type: 'window', includeUncontrolled: true }).then((clientList) => {
      // Se ja tem janela aberta com a URL, foca nela
      for (const client of clientList) {
        if (client.url === targetUrl && 'focus' in client) {
          return client.focus();
        }
      }
      // Senao abre nova janela
      if (self.clients.openWindow) {
        return self.clients.openWindow(targetUrl);
      }
    })
  );
});
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser -> /api/push/subscribe (BFF) | CSRF protegido por SameSite=Strict cookie; autenticacao via session |
| BFF -> /v1/push/subscribe (Fastify) | Service token JWT de curta duracao |
| Worker -> Web Push endpoints | VAPID keys apenas em process.env; nunca em logs |
| push_subscriptions table | Contem endpoint e chaves criptograficas do dispositivo |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-06-02-01 | Information Disclosure | push_subscriptions.endpoint | accept | Endpoint e chave do dispositivo — nao e PII diretamente, mas identifica o dispositivo. Acesso restrito ao worker. |
| T-06-02-02 | Spoofing | VAPID identity | mitigate | VAPID_EMAIL identifica o servidor; keys rotacionadas se comprometidas; worker nao aceita requests externos |
| T-06-02-03 | Denial of Service | Web Push spam | mitigate | Idempotency via notification_log previne envio duplicado; preferences permitem desativar canal push |
| T-06-02-04 | Information Disclosure | BullMQ job payload | mitigate | Payload contem apenas IDs; subscricao buscada do DB em runtime pelo consumer |
| T-06-02-05 | Repudiation | 410 Gone cleanup | accept | Subscricoes expiradas sao deletadas automaticamente no primeiro 410 recebido |
</threat_model>
