```typescript
// apps/api/src/worker/scheduler/delay.ts
import { DateTime } from 'luxon';

/**
 * Calcula o delay em milissegundos ate o proximo horario preferido do usuario.
 * Se o horario ja passou hoje, agenda para o mesmo horario amanha.
 *
 * @param recipientTimezone - IANA timezone string (ex: 'America/Sao_Paulo')
 * @param preferredHour - Hora preferida (0-23) no timezone do usuario
 * @returns delay em milissegundos (sempre > 0)
 */
export function computeJobDelay(recipientTimezone: string, preferredHour: number): number {
  const now = DateTime.utc();

  let sendAt = DateTime.now()
    .setZone(recipientTimezone)
    .set({ hour: preferredHour, minute: 0, second: 0, millisecond: 0 });

  // Se o horario ja passou hoje, agendar para amanha no mesmo horario
  if (sendAt <= now) {
    sendAt = sendAt.plus({ days: 1 });
  }

  return sendAt.toMillis() - now.toMillis();
}

/**
 * Retorna o numero de dias ate um aniversario (considerando apenas MM-DD).
 * Retorna 0 se o aniversario e hoje.
 */
export function daysUntilBirthday(birthDate: Date, referenceDate: Date = new Date()): number {
  const ref = DateTime.fromJSDate(referenceDate).startOf('day');
  const birthday = DateTime.fromJSDate(birthDate);

  let next = ref.set({ month: birthday.month, day: birthday.day }).startOf('day');
  if (next < ref) next = next.plus({ years: 1 });

  return Math.round(next.diff(ref, 'days').days);
}
```

```typescript
// apps/api/src/worker/scheduler/birthday-scheduler.ts
import { Queue, Worker } from 'bullmq';
import { DateTime } from 'luxon';
import type { DrizzleDB } from '../../db/index.js';
import { users, friendships, notificationPreferences, notificationLog } from '../../db/schema.js';
import { eq, and, sql, between } from 'drizzle-orm';
import { computeJobDelay, daysUntilBirthday } from './delay.js';
import type { ReminderType } from '../transports/types.js';
import type { BirthdayReminderJobData } from '../consumers/birthday-reminder-email.consumer.js';

const REMINDER_DAYS: Array<{ days: number; type: ReminderType }> = [
  { days: 7, type: 'D7' },
  { days: 3, type: 'D3' },
  { days: 1, type: 'D1' },
  { days: 0, type: 'Dday' },
];

export function createBirthdayScheduler(
  db: DrizzleDB,
  redisConnection: { host: string; port: number }
) {
  const emailQueue = new Queue<BirthdayReminderJobData>('birthday-reminders', { connection: redisConnection });
  const pushQueue = new Queue<BirthdayReminderJobData>('birthday-reminders-push', { connection: redisConnection });

  // Registrar o scheduler para rodar diariamente as 00:00 UTC
  emailQueue.upsertJobScheduler(
    'daily-birthday-scan',
    { pattern: '0 0 * * *' },
    {
      name: 'scan-upcoming-birthdays',
      data: {} as BirthdayReminderJobData, // dados preenchidos pelo worker
    }
  );

  // Worker que executa o scan quando o scheduler dispara
  const schedulerWorker = new Worker(
    'birthday-reminders',
    async (job) => {
      // Apenas processar o job do scheduler (nao os jobs de entrega)
      if (job.name !== 'scan-upcoming-birthdays') return;

      await scanAndEnqueue(db, emailQueue, pushQueue);
    },
    { connection: redisConnection }
  );

  return { schedulerWorker, emailQueue, pushQueue };
}

async function scanAndEnqueue(
  db: DrizzleDB,
  emailQueue: Queue<BirthdayReminderJobData>,
  pushQueue: Queue<BirthdayReminderJobData>
) {
  const today = new Date();
  const scanWindowDays = 8; // escanear 8 dias para cobrir D7 com margem

  // Query: todos os pares (recipient, birthday_person) onde birthday_person
  // tem aniversario nos proximos 8 dias E sao amigos
  // Query SQL nativa para eficiencia (birthday lookup por MM-DD)
  const upcomingBirthdays = await db.execute(sql`
    SELECT
      f.requester_id   AS recipient_id,
      u.id             AS subject_id,
      u.birth_date     AS birth_date,
      r.timezone       AS recipient_timezone,
      r.notification_send_hour AS send_hour
    FROM friendships f
    JOIN users u ON (
      (f.addressee_id = u.id AND f.requester_id != u.id)
      OR (f.requester_id = u.id AND f.addressee_id != u.id)
    )
    JOIN users r ON r.id = f.requester_id
    WHERE f.status = 'accepted'
      AND u.deleted_at IS NULL
      AND r.deleted_at IS NULL
      AND (
        EXTRACT(MONTH FROM u.birth_date) * 100 + EXTRACT(DAY FROM u.birth_date)
        BETWEEN
          EXTRACT(MONTH FROM CURRENT_DATE) * 100 + EXTRACT(DAY FROM CURRENT_DATE)
          AND
          EXTRACT(MONTH FROM (CURRENT_DATE + INTERVAL '8 days')) * 100 + EXTRACT(DAY FROM (CURRENT_DATE + INTERVAL '8 days'))
      )
  `);

  for (const row of upcomingBirthdays.rows as Array<{
    recipient_id: string;
    subject_id: string;
    birth_date: Date;
    recipient_timezone: string;
    send_hour: number;
  }>) {
    const days = daysUntilBirthday(row.birth_date, today);
    const matchingReminder = REMINDER_DAYS.find((r) => r.days === days);
    if (!matchingReminder) continue;

    // Verificar preferencias do usuario para quais advance days ele quer
    const prefs = await db.select().from(notificationPreferences)
      .where(eq(notificationPreferences.userId, row.recipient_id))
      .limit(10);

    const emailPref = prefs.find((p) => p.channel === 'email');
    const pushPref = prefs.find((p) => p.channel === 'push');

    const daysBefore = emailPref?.daysBefore ?? [1, 7];
    // Dday (days=0) sempre enfileirado se canal habilitado; outros verificam daysBefore
    const shouldSend = days === 0 || daysBefore.includes(days);
    if (!shouldSend) continue;

    const delay = computeJobDelay(
      row.recipient_timezone ?? 'UTC',
      row.send_hour ?? 8,
    );

    const jobData: BirthdayReminderJobData = {
      recipientId: row.recipient_id,
      subjectId: row.subject_id,
      reminderType: matchingReminder.type,
      scheduledFor: new Date(Date.now() + delay).toISOString(),
    };

    const jobId = `birthday-${row.recipient_id}-${row.subject_id}-${matchingReminder.type}-${new Date().getFullYear()}`;

    if (emailPref?.enabled !== false) {
      await emailQueue.add('birthday-reminder', jobData, { delay, jobId, attempts: 3, backoff: { type: 'exponential', delay: 5000 } });
    }

    if (pushPref?.enabled !== false) {
      await pushQueue.add('birthday-reminder', jobData, { delay, jobId: `push-${jobId}`, attempts: 3, backoff: { type: 'exponential', delay: 5000 } });
    }
  }
}
```
