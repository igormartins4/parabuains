```typescript
// apps/api/src/modules/feed/feed.service.ts
import { DateTime } from 'luxon';
import { and, eq, or, sql } from 'drizzle-orm';
import type { DrizzleDB } from '../../db/client.js';
import { users, friendships } from '@parabuains/db';
import type { BirthdayFeedEntry, BirthdayFeedResponse } from './feed.types.js';

export class FeedService {
  constructor(private db: DrizzleDB) {}

  /**
   * Retorna feed de aniversarios dos proximos 30 dias.
   * timezone: IANA timezone string (ex: 'America/Sao_Paulo')
   *
   * Algoritmo:
   * 1. Calcula "hoje" no timezone do viewer
   * 2. Calcula doy (day-of-year) do range: hoje ate hoje+30
   * 3. Busca amigos aceitos com birth_date no range
   * 4. Separa "hoje" de "upcoming"
   * 5. Para cada entrada, calcula daysUntilBirthday
   */
  async getBirthdayFeed(viewerId: string, timezone: string): Promise<BirthdayFeedResponse> {
    // Valida timezone — cai para UTC se invalido
    const tz = DateTime.now().setZone(timezone).isValid ? timezone : 'UTC';
    const now = DateTime.now().setZone(tz);
    const todayDoy = now.ordinal; // 1-365
    const windowEnd = now.plus({ days: 30 });
    const windowEndDoy = windowEnd.ordinal;

    // Busca amigos aceitos com seus dados de aniversario
    const friends = await this.db
      .select({
        id: users.id,
        username: users.username,
        displayName: users.displayName,
        avatarUrl: users.avatarUrl,
        birthDate: users.birthDate,
        birthYearHidden: users.birthYearHidden,
      })
      .from(friendships)
      .innerJoin(
        users,
        or(
          and(eq(friendships.requesterId, viewerId), eq(users.id, friendships.addresseeId)),
          and(eq(friendships.addresseeId, viewerId), eq(users.id, friendships.requesterId)),
        ),
      )
      .where(
        and(
          eq(friendships.status, 'accepted'),
          or(eq(friendships.requesterId, viewerId), eq(friendships.addresseeId, viewerId)),
        ),
      );

    const today: BirthdayFeedEntry[] = [];
    const upcoming: BirthdayFeedEntry[] = [];

    for (const friend of friends) {
      const bd = DateTime.fromJSDate(friend.birthDate).setZone(tz);
      // Construir "proximo aniversario" no ano correto (pode ser ano que vem)
      let nextBirthday = bd.set({ year: now.year });
      if (nextBirthday < now.startOf('day')) {
        nextBirthday = nextBirthday.plus({ years: 1 });
      }
      const diff = nextBirthday.startOf('day').diff(now.startOf('day'), 'days').days;

      if (diff > 30) continue; // fora do range

      const entry: BirthdayFeedEntry = {
        userId: friend.id,
        username: friend.username,
        displayName: friend.displayName,
        avatarUrl: friend.avatarUrl,
        daysUntilBirthday: Math.round(diff),
        birthDate: `${String(bd.month).padStart(2, '0')}-${String(bd.day).padStart(2, '0')}`,
        birthYear: friend.birthYearHidden ? undefined : bd.year,
      };

      if (diff === 0) {
        today.push(entry);
      } else {
        upcoming.push(entry);
      }
    }

    upcoming.sort((a, b) => a.daysUntilBirthday - b.daysUntilBirthday);

    return { today, upcoming, timezone: tz };
  }
}
```

```typescript
// apps/api/src/infrastructure/redis/notification-emitter.ts
import type { Redis } from 'ioredis';
import type { Queue } from 'bullmq';
import type { INotificationEmitter } from '../../modules/friendships/friendship.service.js';

export class RedisNotificationEmitter implements INotificationEmitter {
  constructor(
    private redis: Redis,
    private notificationsQueue: Queue,
  ) {}

  emit(event: string, payload: Record<string, unknown>): void {
    // Fire and forget — nao bloqueia o request
    this.dispatchAsync(event, payload).catch((err) => {
      // Log silencioso — falha de notificacao nao deve derrubar request principal
      console.error('[NotificationEmitter] dispatch error:', err);
    });
  }

  private async dispatchAsync(event: string, payload: Record<string, unknown>): Promise<void> {
    switch (event) {
      case 'friendship.requested': {
        // Publica SSE event no canal Redis do destinatario
        const channel = `notifications:${payload.targetId}`;
        await this.redis.publish(
          channel,
          JSON.stringify({
            type: 'friend_request',
            from: { userId: payload.requesterId },
            friendshipId: payload.friendshipId,
            timestamp: new Date().toISOString(),
          }),
        );
        break;
      }
      case 'friendship.accepted': {
        // Enfileira job BullMQ — Phase 6 implementa o envio de email
        // Anti-pattern 3: NUNCA enviar email inline. Apenas enfileira.
        await this.notificationsQueue.add(
          'friendship.accepted',
          {
            // Apenas IDs — nunca PII (Anti-Pattern 4)
            recipientId: payload.requesterId,
            acceptedById: payload.acceptedById,
            friendshipId: payload.friendshipId,
          },
          {
            attempts: 3,
            backoff: { type: 'exponential', delay: 1000 },
            removeOnComplete: true,
          },
        );
        break;
      }
      default:
        // Evento desconhecido — log para debug
        console.warn('[NotificationEmitter] unknown event:', event);
    }
  }
}
```

Criar Route Handler Next.js para SSE. O endpoint:
1. Autentica via sessao (rejeita com 401 se nao autenticado)
2. Cria Redis subscriber para o canal `notifications:{userId}`
3. Retorna ReadableStream com Content-Type: text/event-stream
4. Envia ping de keepalive a cada 30 segundos
5. Cleanup: cancela o subscriber e o interval quando o cliente desconecta

```typescript
// apps/web/app/api/sse/notifications/route.ts
import { auth } from '@/lib/auth';
import { getRedisSubscriber } from '@/lib/redis';

export const dynamic = 'force-dynamic';

export async function GET(req: Request) {
  const session = await auth.api.getSession({ headers: req.headers });
  if (!session?.user?.id) {
    return new Response('Unauthorized', { status: 401 });
  }

  const userId = session.user.id;
  const channel = `notifications:${userId}`;

  // Cada SSE connection precisa de uma conexao Redis separada (subscribe e exclusivo)
  const subscriber = await getRedisSubscriber();

  const stream = new ReadableStream({
    async start(controller) {
      const encoder = new TextEncoder();

      const send = (data: string) => {
        try {
          controller.enqueue(encoder.encode(data));
        } catch {
          // Stream fechado — ignora
        }
      };

      // Keepalive a cada 30s para evitar timeout de proxy/load balancer
      const keepaliveInterval = setInterval(() => {
        send(': ping\n\n');
      }, 30_000);

      // Subscribe ao canal Redis do usuario
      await subscriber.subscribe(channel, (message) => {
        send(`data: ${message}\n\n`);
      });

      // Cleanup ao fechar
      req.signal.addEventListener('abort', async () => {
        clearInterval(keepaliveInterval);
        await subscriber.unsubscribe(channel);
        await subscriber.quit();
        try { controller.close(); } catch { /* ja fechado */ }
      });

      // Evento inicial de conexao estabelecida
      send(`data: ${JSON.stringify({ type: 'connected', userId })}\n\n`);
    },
  });

  return new Response(stream, {
    headers: {
      'Content-Type': 'text/event-stream',
      'Cache-Control': 'no-cache, no-transform',
      'Connection': 'keep-alive',
      'X-Accel-Buffering': 'no', // desabilita buffering no nginx
    },
  });
}
```

```typescript
// apps/web/hooks/use-sse-notifications.ts
'use client';
import { useEffect, useRef, useState } from 'react';

export interface SseNotification {
  type: 'friend_request' | 'connected';
  from?: { userId: string };
  friendshipId?: string;
  timestamp?: string;
}

export function useSseNotifications() {
  const [notifications, setNotifications] = useState<SseNotification[]>([]);
  const esRef = useRef<EventSource | null>(null);

  useEffect(() => {
    const es = new EventSource('/api/sse/notifications');
    esRef.current = es;

    es.onmessage = (event) => {
      try {
        const data = JSON.parse(event.data) as SseNotification;
        if (data.type === 'connected') return; // ignora evento de conexao
        setNotifications((prev) => [...prev, data]);
      } catch {
        // Ignora mensagens malformadas (ex: ping)
      }
    };

    es.onerror = () => {
      // EventSource reconecta automaticamente — sem acao necessaria
    };

    return () => {
      es.close();
    };
  }, []);

  const clearNotification = (index: number) => {
    setNotifications((prev) => prev.filter((_, i) => i !== index));
  };

  return { notifications, clearNotification };
}
```
